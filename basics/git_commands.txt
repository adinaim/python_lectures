sudo apt install git - установка системы контроля версий

git --version - проверка версии

git config --global user.email "your_email"
git config --global user.name "your name"

git config --list - проверка настроек

ssh-keygen - создали ssh ключ

cat ~/.ssh/id_rsa.pub - получение ssh ключа 

создаем репозитории на github

переходим в папку с нашим проектом 
git init - превращает текущую директорию в git репозиторий
git remote add название связи (обычно - origin) ссылку_на_удаленный_репозиторий(SSH: git@github:....)
git remote -v - просмотр списка удаленных репозиториев (какой репозиторий связан с нашим локальным репозиторием)
git remote rm (обычно - origin) - удаление связи с удаленным репозиторием
git status - проверка статуса файлов
git add название_файлов - отслеживание конкретных файлов
git add . - отслеживание  - отслеживание всех файлов

git commit - создание коммита(версии/слепка кода) 
git commit -m "описание версии" - создание коммита с комментарием

git push название_связи (обычно - origin) название_ветки - отправка версии кода на удаленный репозиторий
Например: git push origin master

git clone ссылка_на_удаленный_репозиторий - склонировать/скопировать/скачать удаленный репозиторий

git diff - просмотр внесенных изменений

git branch - вывод списка веток в терминал
(звездочкой помечается активная ветка)

git branch название_ветки - создание указанной ветки 

git checkout название_ветки
git switch название_ветки   -  переход на указанную ветку

 git pull название_связи название_ветки - стягивание изменнеий с удаленного репозитория, с указанной ветки


git stash - спрятать локальные данные во временную область
git stash list - просмотр списка спрятанных данных
git stash apply - запись спрятанных данных  / извлечение спрятанных данных
git stash drop - удаление спрятанных данных
git log - просмотр коммитов / версий
git log --oneline - компактное отображение коммитов / версий
